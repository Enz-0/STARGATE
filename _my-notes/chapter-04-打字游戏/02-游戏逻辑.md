## 📚 本节内容

### 随机单词生成

```javascript
// 单词列表
const words = [
    'apple', 'banana', 'cherry', 'dragon',
    'elephant', 'flower', 'guitar', 'house'
];

// 随机选择单词
function getRandomWord() {
    const index = Math.floor(Math.random() * words.length);
    return words[index];
}
```

### 计时器实现

```javascript
let timeLeft = 60;
let timerId = null;

function startTimer() {
    timerId = setInterval(() => {
        timeLeft--;
        updateDisplay();
        
        if (timeLeft <= 0) {
            endGame();
        }
    }, 1000);
}

function endGame() {
    clearInterval(timerId);
    alert('游戏结束！');
}
```

### 分数统计

```javascript
let score = 0;
let correctCount = 0;
let wrongCount = 0;

function checkInput(input, target) {
    if (input === target) {
        score += 10;
        correctCount++;
        return true;
    } else {
        wrongCount++;
        return false;
    }
}
```

### 游戏状态管理

```javascript
const gameState = {
    isPlaying: false,
    score: 0,
    timeLeft: 60,
    currentWord: '',
    typedWord: ''
};

function startGame() {
    gameState.isPlaying = true;
    gameState.score = 0;
    gameState.timeLeft = 60;
    gameState.currentWord = getRandomWord();
    startTimer();
}
```
